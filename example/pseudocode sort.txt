INSERTION SORT
deklarasi
	i,j,temp ,n : int
	array : int[]
endDeklarasi
algoritma
	for (i:=1 to i:=n) do
		for (j:=i to j>0 and array[j-1]<array[j]) do
			temp=array[j-1]
			array[j-1]=array[j]
			array[j]=temp
		endFor
	endFor	
endAlgoritma

BUBBLE SORT
deklarasi
	i,j,temp ,n : int
	array : int[]
endDeklarasi
algoritma
	for(i:=0 to i:=n-2) do
		for (j:=0 to j:=n-i-2) do
			if (array[j]<array[j+1]) then
				temp=array[j]
				array[j]=array[j-1]
				array[j-1]=temp
			endIf
		endFor
	endFor
endAlgoritma
